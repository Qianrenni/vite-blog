### 示例对比

```js
// Object
const obj = {};
obj['key'] = 'value';
obj[123] = 'number key'; // 会被转为字符串 '123'

// Map
const map = new Map();
map.set({}, 'value'); // 键可以是对象
map.set(42, 'number key'); // 原始值也可以

// WeakMap
const wm = new WeakMap();
const keyObj = {};
wm.set(keyObj, 'secret');
// wm.set('string', 'value'); // ❌ 报错！键必须是对象
```
