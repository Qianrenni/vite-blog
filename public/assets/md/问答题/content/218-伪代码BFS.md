### 伪代码（BFS）

```python
from collections import deque, Counter

def findMinStep(board, hand):
    def remove_consecutive(s):
        changed = True
        while changed:
            changed = False
            i = 0
            new_parts = []
            n = len(s)
            while i < n:
                j = i
                while j < n and s[j] == s[i]:
                    j += 1
                if j - i >= 3:
                    changed = True
                else:
                    new_parts.append(s[i:j])
                i = j
            s = ''.join(new_parts)
        return s

    hand_count = Counter(hand)
    queue = deque()
    queue.append((board, hand_count, 0))
    visited = set()
    visited.add((board, tuple(sorted(hand_count.items()))))

    while queue:
        curr_board, curr_hand, steps = queue.popleft()
        if not curr_board:
            return steps

        # 尝试每种可用字符
        for char in list(curr_hand.keys()):
            if curr_hand[char] == 0:
                continue
            # 尝试每个插入位置（优化：只插在相同字符边界）
            for i in range(len(curr_board) + 1):
                # 优化：避免无效插入
                # 例如：不在已有 char 块附近插入，可能浪费
                if i > 0 and curr_board[i-1] == char:
                    pass  # 可插
                elif i < len(curr_board) and curr_board[i] == char:
                    pass
                elif i > 0 and i < len(curr_board) and curr_board[i-1] == curr_board[i]:
                    pass  # 可能形成新组合
                else:
                    # 简单起见，也可以不剪枝，但效率低
                    pass

                new_board = curr_board[:i] + char + curr_board[i:]
                new_board = remove_consecutive(new_board)

                new_hand = curr_hand.copy()
                new_hand[char] -= 1
                if new_hand[char] == 0:
                    del new_hand[char]

                state = (new_board, tuple(sorted(new_hand.items())))
                if state not in visited:
                    visited.add(state)
                    queue.append((new_board, new_hand, steps + 1))

    return -1  # 无法消除
```
