## JS变量提升,事件执行顺序代码考察
```javascript

function foo(){
  console.log(this.bar);
}

var bar = 'bar';
var o1 = {bar:'bar1', foo:foo};
var o2 = {bar:'bar2', foo:()=>foo()};
foo();
o1.foo();
o2.foo();
```

```javascript
var nam = 'world';
(function(){
  if(typeof nam==='undefined'){
    nam ='Jack';
    console.log(`Hello ${nam}`);
  }else{
    console.log(`Hello ${nam}`);
  }
})();
```
```javascript
function Person(){
```
